	if     rme96 rcreg   RME96 RCR IRQ  ||
	       rme96 rcreg   RME96 RCR IRQ 2   
	{
		return IRQ NONE 
	}
	
	if  rme96 rcreg   RME96 RCR IRQ  {
		   playback

	}

}
